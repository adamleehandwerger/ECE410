\section*{1. Expansion via Binomial Theorem}
 
Applying the binomial theorem with $p = 3$:
\[
  K(x,y) = (c + x^\top y)^{3}
          = c^{3} + 3c^{2}(x^\top y)
            + 3c(x^\top y)^{2} + (x^\top y)^{3}.
\]
Each power of $x^\top y$ lifts to a tensor inner product.
Writing $\langle A, B \rangle_{F} = \mathrm{tr}(A^\top B)$ for the
Frobenius inner product:
\[
  (x^\top y)^{k} = \langle x^{\otimes k},\, y^{\otimes k} \rangle,
\]
where $x^{\otimes k} = x \otimes \cdots \otimes x$ ($k$ times) is a rank-1
symmetric tensor of order $k$.

\section*{2. Feature Map}
 
The explicit feature map $\phi : \mathbb{R}^{d} \to \mathcal{H}$ is
\[
  \phi(x)
  = \begin{pmatrix}
      c^{3/2} \\[4pt]
      \sqrt{3}\,c\; x \\[4pt]
      \sqrt{3c}\; x^{\otimes 2} \\[4pt]
      x^{\otimes 3}
    \end{pmatrix},
\]
where $x^{\otimes 2} = xx^\top \in \mathbb{R}^{d \times d}$ and
$x^{\otimes 3} \in \mathbb{R}^{d \times d \times d}$, both flattened.
Verification:
\[
  \langle \phi(x), \phi(y) \rangle
  = c^{3} + 3c^{2}(x^\top y) + 3c(x^\top y)^{2} + (x^\top y)^{3}
  = K(x,y). \qquad \checkmark
\]

\section*{3. The RKHS as a Tensor Space}
 
Every $f \in \mathcal{H}_{K}$ is parametrised by
$(a,\, b,\, C,\, \mathcal{T})$ with
\[
  a \in \mathbb{R},\quad
  b \in \mathbb{R}^{d},\quad
  C \in \mathbb{R}^{d \times d},\quad
  \mathcal{T} \in \mathbb{R}^{d \times d \times d},
\]
and evaluated as
\[
  \boxed{
    f(x) = a + b^\top x
           + \mathrm{tr}(C^\top x x^\top)
           + \langle \mathcal{T},\, x^{\otimes 3} \rangle.
  }
\]
The cubic term satisfies
$\langle \mathcal{T},\, x^{\otimes 3} \rangle
 = x^\top \mathcal{T}_{(1)}(x \otimes x)$,
where $\mathcal{T}_{(1)} \in \mathbb{R}^{d \times d^{2}}$ is the
mode-1 matricisation of $\mathcal{T}$.
 
% ---------------------------------------------------------------
\section*{4. RKHS Inner Product}
 
For $f \leftrightarrow (a, b, C, \mathcal{T})$ and
$g \leftrightarrow (a', b', C', \mathcal{T}')$:
\[
  \boxed{
    \langle f,\, g \rangle_{\mathcal{H}}
    = \frac{aa'}{c^{3}}
    + \frac{b^\top b'}{3c^{2}}
    + \frac{\langle C,\, C' \rangle_{F}}{3c}
    + \langle \mathcal{T},\, \mathcal{T}' \rangle_{F},
  }
\]
where
$\langle \mathcal{T}, \mathcal{T}' \rangle_{F}
 = \mathrm{tr}(\mathcal{T}_{(1)}^\top \mathcal{T}'_{(1)})$.
The RKHS norm is
\[
  \| f \|_{\mathcal{H}}^{2}
  = \frac{a^{2}}{c^{3}}
  + \frac{\| b \|^{2}}{3c^{2}}
  + \frac{\| C \|_{F}^{2}}{3c}
  + \| \mathcal{T} \|_{F}^{2}.
\]
 
% ---------------------------------------------------------------
\section{5. Representer in Tensor Form}
 
The kernel section $K(\cdot, x)$ corresponds to
\[
  K(\cdot, x)
  \;\leftrightarrow\;
  \bigl(\,c^{3},\quad 3c^{2} x,\quad 3c\, xx^\top,\quad x^{\otimes 3}\,\bigr).
\]
The reproducing property:
\begin{align*}
  \langle f,\, K(\cdot, x) \rangle_{\mathcal{H}}
  &= \frac{a \cdot c^{3}}{c^{3}}
   + \frac{b^\top(3c^{2} x)}{3c^{2}}
   + \frac{\langle C,\, 3c\, xx^\top \rangle_{F}}{3c}
   + \langle \mathcal{T},\, x^{\otimes 3} \rangle_{F} \\[4pt]
  &= a + b^\top x + x^\top C x
   + \langle \mathcal{T},\, x^{\otimes 3} \rangle
   \;=\; f(x). \qquad \checkmark
\end{align*}
 
% ---------------------------------------------------------------
\section*{6. Kernel Matrix (No Loops)}
 
On a dataset $X = [x_{1} \mid \cdots \mid x_{n}]
\in \mathbb{R}^{d \times n}$, let
$\mathbf{G} = c\,\mathbf{1}\mathbf{1}^\top + X^\top X$.
The full kernel matrix is the Hadamard cube:
\[
  \boxed{\mathbf{K} = \mathbf{G} \circ \mathbf{G} \circ \mathbf{G}.}
\]
Assembled from one matrix multiply and two elementwise products---no
loops required.

\section*{7. General Degree-$p$ Pattern}
 
For $K(x,y) = (c + x^\top y)^{p}$:
\[
  K(x,y)
  = \sum_{k=0}^{p} \binom{p}{k} c^{p-k}
    \langle x^{\otimes k},\, y^{\otimes k} \rangle.
\]

$1$ & $x$
    & $\displaystyle \left[\binom{p}{1}c^{p-1}\right]^{-1}$
    & $b \in \mathbb{R}^{d}$ \\[10pt]

    & $\displaystyle \left[\binom{p}{2}c^{p-2}\right]^{-1}$
    & $C \in \mathbb{R}^{d \times d}$ \\[10pt]
$\vdots$ & $\vdots$ & $\vdots$ & $\vdots$ \\[3pt]
$k$ & $x^{\otimes k}$
    & $\displaystyle \left[\binom{p}{k}c^{p-k}\right]^{-1}$
    & $\mathcal{T}^{(k)} \in \mathbb{R}^{d^{k}}$ \\[10pt]

    & $\mathcal{T}^{(p)} \in \mathbb{R}^{d^{p}}$ \\

\medskip
The RKHS is the space of \textbf{polynomials of degree $\leq p$} on
$\mathbb{R}^{d}$, with geometry determined by symmetric tensor algebra
and Frobenius inner products---no index gymnastics needed.
